131  void ErrorEuclidian( Vect<double>& x, Vect<double>& y )
132  {
133  vops::Dot<double> dot;
134  Vect<double> err( y.Size() );
135  for ( ULong i = 0; i < err.Size(); ++i )
136  {
137  err[i] = operator()( x[i] ) - y[i];
138  }
139  m_error = sqrt( dot( err, err ) ) / static_cast<double>( err.Size() );
140  }
