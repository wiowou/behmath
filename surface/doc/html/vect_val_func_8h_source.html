138  void Solve()
139  {
140  ULong size = m_comp.Size();
141 
142  Vect<double> b = *m_b; //copying the original b
143  Vect<double>& fx = *m_b;
144  Vect<double>& x = *m_x;
145  x.Resize( size );
146 
147  Vect<double> xmod( size ); //the modification to x
148  Matrix<Vect> J( size );
149 
150  for ( ULong i = 0; i < size; ++i )
151  {
152  xmod[i] = m_tolerance * 2.0; //to ensure while loop is entered at least once
153  }
154 
155  vops::Subtr<double> subtr;
156  solver::Elimination<Vect> solver;
157  solver.Tolerance(m_tolerance);
158  solver.PartialPivoting(true);
159 
160  m_iter = 0;
161  while ( m_iter < m_maxIter && Continue(xmod) )
162  {
163  operator()(x); //finding fx
164 
165  subtr( fx, b, xmod );
166 
167  xmod.Swap(fx);
168  //create the Jacobian
169  for ( ULong i = 0; i < size; ++i )
170  {
171  m_comp[i]->Grad( x, J[i] );
172  }
173 
174  solver.A(J);
175  solver.B(fx);
176  solver.X(xmod);
177  solver.Solve();
178 
179  subtr( x, xmod, x );
180 
181  ++m_iter;
182  }
183 
184  m_b->Swap(b); //replacing b
185  }
186 protected:
187  inline bool Continue( Vect<double>& xmod )
188  {
189  //this can be parallelized
190  for ( ULong i = 0; i < xmod.Size(); ++i )
191  {
192  if ( std::abs( xmod[i] ) > m_tolerance )
193  {
194  return true;
195  }
196  }
197  return false;
198  }
