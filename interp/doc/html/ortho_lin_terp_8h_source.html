92  double Interp( const double v[], const VecVec &vv, bool calcJumpSizes )
93  {
94  //finding the indexes of the lower and upper bound for each independent paramater, v[i]
95  //find the natural coordinate for each independent variable
96  for ( int i = 0; i < ndim; ++i )
97  {
98  m_up = std::upper_bound( vv[i].begin(), vv[i].end(), v[i] ); // <algorithm>
99  if ( m_up == vv[i].begin() )
100  {
101  m_idx[0][i] = m_idx[1][i] = 0;
102  m_eta[i] = -1.0;
103  }
104  else if ( m_up == vv[i].end() )
105  {
106  m_idx[0][i] = m_idx[1][i] = vv[i].size() - 1;
107  m_eta[i] = 1.0;
108  }
109  else
110  {
111  m_idx[1][i] = m_up - vv[i].begin();
112  m_idx[0][i] = m_idx[1][i] - 1;
113  //multiplying by 2 and subtracting 1 converts the ratio to natural coordinates
114  m_eta[i] = 2.0 * ( v[i] - vv[i][ m_idx[0][i] ] ) / ( vv[i][ m_idx[1][i] ] - vv[i][ m_idx[0][i] ] ) - 1.0;
115  }
116  }
117  // done finding bounds, natural coordinates
118 
119  //convenience calc to determine the size of the jumps needed to navigate the dependent variable vector, vv[ndim]
120  if ( calcJumpSizes )
121  {
122  CalcJumpSizes(vv);
123  }
124  // done finding jump sizes
125 
126  //calculate the 2^ndim shape functions. Each function has ndim coordinates that are multiplied together
127  // formula for the shape functions in n dimensions comes from NatCoordShapeFunc.pdf, page 11-5, formula 11.3
128  double result = 0.0;
129  for ( int i = 0; i < m_pow2ndim; ++ i )
130  {
131  std::size_t b[ndim];
132  ToBitArray( i, b );
133  std::size_t resIdx = 0;
134  double shapeFuncVal = 1.0;
135  for ( int k = 0; k < ndim; ++k )
136  {
137  resIdx += m_idx[ b[k] ][k] * m_jmp[k]; //b[k] should resolve to 0 for false and 1 for true if the bit is set.
138  shapeFuncVal *= 1.0 + ( 2.0 * static_cast<double>(static_cast<int>( b[k] ) ) - 1.0 ) * m_eta[k];
139  }
140  shapeFuncVal *= vv[ndim][resIdx];
141  result += shapeFuncVal;
142  }
143  return result * m_shapeFuncCoeff;
144  }

178  double Interp1( double v, const VecVec &vv )
179  {
180  m_up = std::upper_bound( vv[0].begin(), vv[0].end(), v );
181  if ( m_up == vv[0].begin() )
182  {
183  return vv[1].front();
184  }
185  if ( m_up == vv[0].end() )
186  {
187  return vv[1].back();
188  }
189  std::size_t ind = m_up - vv[0].begin();
190  double yr = ( v - vv[0][ind - 1] ) / ( vv[0][ind] - vv[0][ind - 1] );
191  return yr * ( vv[1][ind] - vv[1][ind - 1] ) + vv[1][ind - 1];
192  }
