286  inline void RowReduce( T factor, ULong reducingRow, ULong reducedRow ) //tested
287  {

303  inline void RowSwap( ULong rowNum1, ULong rowNum2 )
304  {
305  if ( rowNum1 == rowNum2 )
306  {
307  return;
308  }
309  ULong rowNum1idx = Find(rowNum1);

311 
312  if ( rowNum1idx == m_nnz && rowNum2idx == m_nnz )
313  {
314  return;
315  }
316 
317  if ( rowNum1idx == m_nnz )

319  Insert( rowNum1, m_data[rowNum2idx] );
320  m_data[rowNum2idx] = T();

322  }
323 
324  if ( rowNum2idx == m_nnz )
325  {
326  Insert( rowNum2, m_data[rowNum1idx] );
327  m_data[rowNum1idx] = T();
328  return;
329  }

331  T tmp = m_data[rowNum1idx];
332  m_data[rowNum1idx] = m_data[rowNum2idx];
333  m_data[rowNum2idx] = tmp;

372  ULong UpperBoundIdx( const ULong val ) const //tested
373  {
374  ULong const* beg = m_pos;
375  ULong const* end = &m_pos[m_nnz];
376  if ( val <= *beg )
377  {
378  return 0;
379  }
380  --end;
381  if ( val > *end )
382  {
383  return ( end - beg ) + 1;
384  }
385  ULong const* zeroPos = beg;
386  while ( ( end - beg ) > 2 )
387  {
388  //ULong const* mid = &beg[ ( end - beg ) >> 1 ]; // >> 1 is division by 2
389  double pos = ( static_cast<double>(val) - static_cast<double>(*beg) ) /
390  ( static_cast<double>(*end) - static_cast<double>(*beg) ) * static_cast<double>(end - beg);
391  double fraction = pos - static_cast<ULong>(pos);
392  if ( fraction < 0.001 )
393  {
394  pos += 0.01;
395  }
396  ULong const* mid = &beg[ static_cast<ULong>(pos) ];
397  if ( *mid == val )
398  {
399  return (mid - zeroPos);
400  }
401  if ( mid == beg )
402  {
403  ++mid;
404  }
405  else if ( mid == end )
406  {
407  --mid;
408  }
409  if ( *mid > val )
410  {
411  end = mid;
412  }
413  else
414  {
415  beg = mid;
416  }
417  }
418  ULong dist = end - beg;
419  for ( ULong i = 0; i <= dist; ++i )
420  {
421  if ( *beg >= val )
422  {
423  return (beg - zeroPos);
424  }
425  ++beg;
426  }
427  }

431  void Set( ULong idx, T val, ULong ub ) //tested
432  {
433  if ( m_nnz == 0 )

436  m_data[0] = val;

443  }
444  if ( ub < m_nnz && m_pos[ub] == idx )
445  {
446  m_data[ub] = val;
447  return;
448  }
449  if ( m_nnz == m_allocSize - 1 )
450  {
451  m_allocSize = 2 * m_allocSize;
452  ULong* pos = new ULong [m_allocSize];
453  T* data = new T [m_allocSize];
454  for ( ULong i = 0; i < ub; ++i)
455  {
456  pos[i] = m_pos[i];
457  data[i] = m_data[i];
458  }
459  pos[ub] = idx;
460  data[ub] = val;
461  for ( ULong i = ub + 1; i < m_nnz + 1; ++i )
462  {
463  pos[i] = m_pos[i - 1];
464  data[i] = m_data[i - 1];
465  }
466  ULong* tmp = m_pos;
