241  ULong MaxIdxBelow( const ULong pivotIdx ) const //tested
242  {
243  ULong rowEnd = m_size;
244  ULong max = pivotIdx;
245  T maxVal = m_row[pivotIdx][pivotIdx];
246  for ( ULong i = pivotIdx + 1; i < rowEnd; ++i )
247  {
