129  void Solve()
130  {
131  // Ax = b
132  // (Q*R)x = b, Q = Q^T = Q^-1
133  // R*x = Q^T * b
134  // R is upper triangular, back solve up for x
135  Matrix< Vect, T >& R = *m_A;
136  //Vect<T>& b = *m_b;
137  Vect<T>& x = *m_x;
138

156  ULong Eig( ULong maxIter, T tol = 1e-5, bool calcEigVect = false )
157  {
158  Matrix< Vect, T >& R = *m_A;
159  Matrix< Vect, T >& A = *m_A;
160  Matrix< Vect, T >& evect = m_EigVect;
161  Matrix< Vect, T >& Q = m_QTr;
162  Matrix<SparseVect,T>& eval = m_EigVal;
163 
164  if ( !m_factorized )
165  {
166  Factorize();
167  }
168 
169  eval.Clear();
170  if ( calcEigVect )
171  {
172  evect.Resize( Q.Rows() );
173  //Transpose the eigenvector matrix for easy multiplication
174  for ( ULong i = 0; i < Q.Rows(); ++i )
175  {
176  for ( ULong j = 0; j < Q.Rows(); ++j )
177  {
178  evect[i][j] = Q[j][i];
179  }
180  }
181  }
182 
183  vops::Dot<T> dot;
184  Matrix< Vect, T > AP( Q.Rows() );
185  dot( R, Q, AP );
186 
187  ULong iter = 0;
188  while ( iter < maxIter && ExceedsOffDiagTol( AP, tol) )
189  {
190  A.Swap( AP );
191  Factorize();
192  if ( calcEigVect )
193  {
194  dot( evect, Q, AP ); //use A' as temporary storage of the matrix result
195  evect.Swap(AP); //swap result into the eigenvector matrix
196  }
197  dot( R, Q, AP );
198  ++iter;
199  }
200 
201  if( iter < maxIter )
202  {
203  eval.Resize( AP.Rows() );
204  for ( ULong i = 0; i < AP.Rows(); ++i )
205  {
206  eval[i].PushBack( i, AP[i][i] );
207  }
208  }
209  return iter;
210  }

228  void HouseHolderQn_x_( Vect<T>& v, ULong n )
229  {
230  Matrix< Vect, T >& A = *m_A;
231  vops::Dot<T> dot;
232 
233  T qu = dot( A[n], v, n );
234  T uuRecip = -2.0 / dot( A[n], A[n], n ); //multiply by the -2.0 and combine with reciprocal
