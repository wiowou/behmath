84  void Solve()
85  {
86  vops::Dot<double> dot;
87  double resultRK1;
88  double resultRK2 = m_initialCondition;
89  double lastResult = m_initialCondition;
90  double dt = m_tFinal - m_tInitial;
91  double dtDouble = 0.1 * m_tolerance;
92  int dtDoubleCounter = 0;
93  m_steps = 0;
94  Vect<double> x(2);
95  double epsilon = ( m_tFinal - m_tInitial ) / static_cast<double>( m_maxSteps * 100 );
96  Vect<double> k(2);
97 
98  Vect<double> bRK1(2);
99  Vect<double> bRK2(2);
100  bRK1[0] = 1.0; bRK1[1] = 0.0;
101  bRK2[0] = 0.5; bRK2[1] = 0.5;
102 
103  m_time = m_tFinal;
104  while ( m_time < m_tFinal + epsilon && m_steps < m_maxSteps )
105  {
106  x[1] = m_time;
107 
108  x[0] = resultRK2;
109  k[0] = (*m_f)(x);
110 
111  x[0] = resultRK2 + k[0] * dt;
112  k[1] = (*m_f)(x);
113 
114  resultRK1 = resultRK2 + dot( k, bRK1 ) * dt;
115  lastResult = resultRK2;
116  resultRK2 = resultRK2 + dot( k, bRK2 ) * dt;
117 
118  double error = std::abs( resultRK2 - resultRK1 );
119  //Adaptive time-stepping
120  if ( error < m_tolerance )
121  {
122  if ( !ResultInRange(resultRK2) )
123  {
124  m_result = lastResult;
125  resultRK2 = lastResult;
126  m_time -= dt;
127  dt *= 0.5;
128  if ( std::abs( m_result - m_fLow ) < m_rangeTol || std::abs( m_result - m_fUp ) < m_rangeTol )
129  {
130  return;
131  }
132  }
133  //The dtDoubleCounter prevents successive doubling and halving of the time step
134  if ( error < dtDouble && dtDoubleCounter > 16 )
135  {
136  dt *= 2.0;
137  dtDoubleCounter = 0;
138  }
139  }
140  else
141  {
142  m_time -= dt;
143  resultRK2 = lastResult;
144  dt *= 0.5;
145  }
146 
147  if ( m_time + dt > m_tFinal )
148  {
149  dt = m_tFinal - m_time < epsilon ? epsilon : m_tFinal - m_time;
150  }
151  m_time += dt;
152  ++dtDoubleCounter;
153  ++m_steps;
154  }
155  m_result = resultRK2;
156  }
157 
158 protected:
159  bool ResultInRange( double f )
160  {
161  return ( f > m_fLow && f < m_fUp );
162  }

170  }
