37 template<class F, class T = double>
38 class InvQuad : public RootFinder<F,T>
39 {
40  using RootFinder<F,T>::m_f;
41  using RootFinder<F,T>::m_tolerance;
42  using RootFinder<F,T>::m_x;
43  using RootFinder<F,T>::m_root;
44  using RootFinder<F,T>::m_maxIter;
45  using RootFinder<F,T>::m_iter;
46  using RootFinder<F,T>::m_runCalc;
47  using RootFinder<F,T>::m_rootFound;
48 
49 public:
50  InvQuad()
51  {
52  m_A.Resize(3);
53  m_A[0][0] = 1.0;
54  m_A[1][0] = 1.0;
55  m_A[2][0] = 1.0;
56  m_x.Resize(3);
57  m_b.Resize(3);
58  }
59 
60  T Root()
61  {
62  if ( m_runCalc )
63  {
64  Calc();
65  }
66  return m_root;
67  }
68 
69 protected:
70  void Calc()
71  {
72  Matrix<Vect,T>& A = m_A;
73  Vect<T>& x = m_x;
74  Vect<T>& b = m_b;
75 
76  A[0][1] = m_f(x[0]);
77  A[1][1] = m_f(x[1]);
78  A[2][1] = m_f(x[2]);
79  A[0][2] = A[0][1] * A[0][1];
80  A[1][2] = A[1][1] * A[1][1];
81  A[2][2] = A[2][1] * A[2][1];
82 
83  solver::Elimination<Vect,T> sol;
84  sol.Tolerance(1e-5);
85  sol.PartialPivoting(true);
86 
87  Matrix<Vect,T> Acpy = A;
88  sol.A(Acpy);
89  Vect<T> xcpy = x;
90  sol.B(xcpy); //the vector of outputs, b, is the known input x
91  sol.X(b); //we are solving for the coefficients, b, of the parabola
92  sol.Solve();
93  m_root = b[0];
94 
95  m_iter = 0;
96  while ( m_iter < m_maxIter && std::abs( m_f( m_root ) ) > m_tolerance )
97  {
98  x[0] = x[1];
99  x[1] = x[2];
100  x[2] = b[0];
101 
102  A[0][1] = A[1][1];
103  A[1][1] = A[2][1];
104  A[2][1] = m_f(x[2]);
105 
106  A[0][2] = A[1][2];
107  A[1][2] = A[2][2];
108  A[2][2] = A[2][1] * A[2][1];
109 
110  Acpy = A;
111  xcpy = x;
112  sol.Solve();
113  m_root = b[0];
114 
115  ++m_iter;
116  }
117  m_runCalc = false;
118  if ( m_iter < m_maxIter )
119  {
120  m_rootFound = true;
121  }
122  }
123 
124 protected:
125  Vect<T> m_b;
126  Matrix<Vect,T> m_A;
127 
128 private:
129

133 #endif //MYDEBUG
134 };
