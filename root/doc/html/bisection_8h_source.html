94  //The main routine to find the root
95  void Calc()
96  {
97  T& low = m_low;
98  T& high = m_high;
99  T& mid = m_root;
100 
101  if ( !RootInRange() )
102  {
103  return;
104  }
105 
106  mid = ( high + low ) / 2.0;
107 
108  //added to allow for a better midpoint
109  T mid2;
110  m_mid.X0(mid);
111  m_mid.X1(high);
112  m_mid.X2(low);
113  mid2 = m_mid.Root();
114  mid = mid2 > high || mid2 < low ? mid : mid2;
115  //
116  T oldRange = ( high - low ) / m_fracTarget * 2.0;
117 
118  m_iter = 0;
119  while ( m_iter < m_maxIter && std::abs( m_f(mid) ) > m_tolerance )
120  {
121  if ( m_f(mid) < T(0) )
122  {
123  low = mid;
124  }
125  else
126  {
127  high = mid;
128  }
129  mid = ( high + low ) / 2.0;
130 
131  // added to make sure the other method is keeping up with the bisection method.
132  // Checks every m_switch iterations and gives hybrid method another chance every m_switch iterations
133  if ( m_iter % m_switch == 0 )
134  {
135  if ( !m_useHybrid )
136  {
137  m_useHybrid = true;
138  }
139  else if ( ( high - low ) > oldRange * m_fracTarget )
140  {
141  m_useHybrid = false;
142  }
143  oldRange = high - low;
144  }
145 
146  if ( m_useHybrid )
147  {
148  //added to allow for a better midpoint
149  T mid2;
150  m_mid.X0(mid);
151  m_mid.X1(high);
152  m_mid.X2(low);
153  mid2 = m_mid.Root();
154  mid = mid2 > high || mid2 < low ? mid : mid2;
155  //
156  }
157 
158  ++m_iter;
159  }
160  m_runCalc = false;
161  if ( m_iter < m_maxIter )
162  {
163  m_rootFound = true;
164  }
165  }
