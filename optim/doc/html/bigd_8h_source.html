98  void Range( Vect<double> &low, Vect<double> &high )
99  {
100  MinFinder<F,Vect<double> >::Range( low, high );
101  m_guess.Resize( low.Size() );
102  m_vel.Resize( low.Size() );
103  m_acc.Resize( low.Size() );
104  for ( ULong i = 0; i < m_guess.Size(); ++i )
105  {
106  m_guess[i] = 0.5 * ( high[i] + low[i] );
107  m_dt = m_dt > 0.1 * ( high[i] - low[i] ) ? 0.1 * ( high[i] - low[i] ) : m_dt;
108  }
109  m_minVal = (*m_f)( m_guess );
110  }
111 
112 protected:
113  void Calc()
114  {
115  Vect<double> &pos = m_guess;
116  Vect<double> &vel = m_vel;
117  Vect<double> &acc = m_acc;
118  double dt = m_dt;
119 
120  m_f->Grad( pos, acc );
121  m_iter = 0;
122  while ( Continue( acc, m_tolerance ) && m_iter < m_maxIter )
123  {
124  for ( ULong i = 0; i < pos.Size(); ++i )
125  {
126  pos[i] -= acc[i] * dt + vel[i] * dt;
127  vel[i] += m_cf * acc[i] * dt;
128  }
129  Bounce( pos, vel );
130 
131  double val = (*m_f)( pos );
132  if ( val < m_minVal )
133  {
134  m_min = pos;
135  m_minVal = val;
136  }
137 
138  m_f->Grad( pos, acc );
139  ++m_iter;
140  }
141 
142  m_runCalc = false;
143  if ( m_iter < m_maxIter )
144  {
145  m_minFound = true;
146  }
147  }

163  void Bounce( Vect<double>& pos, Vect<double>& vel )
164  {
165  for ( ULong i = 0; i < pos.Size(); ++i )
166  {
167  if ( pos[i] < m_low[i] )
168  {
169  pos[i] = m_low[i];
170  vel[i] *= -1.0;
171  }
172  else if ( pos[i] > m_high[i] )
173  {
174  pos[i] = m_high[i];
175  vel[i] *= -1.0;
176  }
177  }
178  }
